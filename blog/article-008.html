1. 获取并清洗期货日线数据


import akshare as ak
import pandas as pd

def get_futures(symbol, start="2020-01-01"):
    df = ak.futures_zh_daily_sina(symbol=symbol)
    df["date"] = pd.to_datetime(df["date"])
    df = df.sort_values("date").reset_index(drop=True)
    df = df[df["volume"] > 0]
    if start:
        df = df[df["date"] >= pd.to_datetime(start)]
    return df

2. 协整检验


from statsmodels.tsa.stattools import coint
import numpy as np

def check_cointegration(price_a, price_b):
    """返回(是否协整, p值, 推荐对冲比)"""
    log_a = np.log(price_a)
    log_b = np.log(price_b)
    
    score, pvalue, _ = coint(log_a, log_b)
    
    # 计算OLS对冲比
    from numpy.polynomial import polynomial as P
    hedge_ratio = np.polyfit(log_b, log_a, 1)[0]
    
    return pvalue < 0.05, pvalue, hedge_ratio

3. 滚动Z-score计算


def rolling_zscore(series, window=20):
    mean = series.rolling(window).mean()
    std = series.rolling(window).std()
    return (series - mean) / std

4. 最大回撤计算


def max_drawdown(returns):
    """传入每日收益率序列（小数形式）"""
    cum = (1 + returns).cumprod()
    peak = cum.expanding().max()
    dd = (cum - peak) / peak
    return dd.min()

5. 夏普比率计算


def sharpe_ratio(returns, rf=0.02, periods=252):
    """returns: 每日收益率, rf: 年化无风险利率"""
    excess = returns - rf/periods
    return np.sqrt(periods) * excess.mean() / returns.std()

6. 布林带计算


def bollinger_bands(series, window=20, std_mult=2):
    ma = series.rolling(window).mean()
    std = series.rolling(window).std()
    return {
        "upper": ma + std_mult * std,
        "middle": ma,
        "lower": ma - std_mult * std
    }

7. ADX趋势强度


def calc_adx(df, period=14):
    """df需要包含high, low, close列"""
    high = df["high"]
    low = df["low"]
    close = df["close"]
    
    tr = pd.concat([
        high - low,
        (high - close.shift()).abs(),
        (low - close.shift()).abs()
    ], axis=1).max(axis=1)
    
    atr = tr.ewm(span=period, adjust=False).mean()
    
    up_move = high - high.shift()
    down_move = low.shift() - low
    
    dm_plus = pd.Series(np.where((up_move > down_move) & (up_move > 0), up_move, 0), index=df.index)
    dm_minus = pd.Series(np.where((down_move > up_move) & (down_move > 0), down_move, 0), index=df.index)
    
    di_plus = 100 * dm_plus.ewm(span=period, adjust=False).mean() / atr
    di_minus = 100 * dm_minus.ewm(span=period, adjust=False).mean() / atr
    
    dx = 100 * (di_plus - di_minus).abs() / (di_plus + di_minus)
    adx = dx.ewm(span=period, adjust=False).mean()
    
    return adx

            timeout=5
        )

        return False

9. 交易绩效统计


def trade_stats(trades_df):
    """trades_df需要包含: pnl（盈亏）列"""
    wins = trades_df[trades_df["pnl"] > 0]
    losses = trades_df[trades_df["pnl"] <= 0]
    
    win_rate = len(wins) / len(trades_df) if len(trades_df) > 0 else 0
    avg_win = wins["pnl"].mean() if len(wins) > 0 else 0
    avg_loss = abs(losses["pnl"].mean()) if len(losses) > 0 else 0
    pnl_ratio = avg_win / avg_loss if avg_loss > 0 else 0
    
    return {
        "总交易数": len(trades_df),
        "胜率": f"{win_rate:.1%}",
        "盈亏比": f"{pnl_ratio:.2f}",
        "总盈亏": f"{trades_df['pnl'].sum():.0f}",
        "平均盈利": f"{avg_win:.0f}",
        "平均亏损": f"-{avg_loss:.0f}"
    }

10. 简单回测框架


def simple_backtest(price, signals, initial_capital=100000, contract_size=10):
    """
    price: 收盘价序列
    signals: 1=多仓 -1=空仓 0=空仓（每日信号）
    """
    capital = initial_capital
    position = 0
    entry_price = 0
    equity = [capital]
    
    for i in range(1, len(price)):
        sig = signals.iloc[i]
        p = price.iloc[i]
        
        if position != sig:  # 信号变化，先平后开
            if position != 0:
                pnl = (p - entry_price) * position * contract_size
                capital += pnl
            
            position = sig
            entry_price = p
        
        # 计算浮动净值
        unrealized = (p - entry_price) * position * contract_size if position != 0 else 0
        equity.append(capital + unrealized)
    
    equity = pd.Series(equity, index=price.index)
    returns = equity.pct_change().dropna()
    
    return {
        "最终净值": equity.iloc[-1],
        "总收益率": f"{(equity.iloc[-1]/initial_capital - 1):.1%}",
        "夏普比率": f"{sharpe_ratio(returns):.2f}",
        "最大回撤": f"{max_drawdown(returns):.1%}"
    }
